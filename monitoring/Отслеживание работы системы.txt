class PerformanceMonitor:
    def __init__(self):
        self.equity_curve = []
    def update(self, pnl: float):
        self.equity_curve.append(pnl)

    def sharpe_ratio(self):
        returns = pd.Series(self.equity_curve).pct_change()
        return np.sqrt(365) * returns.mean() / returns.std()

class AlertSystem:
    def check_anomalies(self, data):
        # Использование Isolation Forest для обнаружения аномалий
        model = IsolationForest(contamination=0.01)
        anomalies = model.fit_predict(data)
        if -1 in anomalies:
            self.send_alert("Обнаружены аномальные данные!")
    def send_alert(self, message):
        # Интеграция с Telegram/Slack
        print(f"[ALERT] {message}")
